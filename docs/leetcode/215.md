# Leetcode 215 数组中的第K大元素

```
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var findKthLargest = function (nums, k) {
  let array = new Array();
  for (let i = 0; i < nums.length; i++) {
    // 如果数组的元素数量超过 k
    if (array.length >= k) {
      // 获取数组中最小的元素
      let smallest = array.min();
      // 如果数组中最小的元素 < nums[i]
      if (smallest < nums[i]) {
        for (let j = 0; j < array.length; j++) {
          if (array[j] == smallest) {
            // 删除最小的元素
            array.splice(j, 1);
            break;
          }
        }
        // 往数组中写入 nums[i]
        array.push(nums[i]);
      }
    } else {
      // 往数组中写入 nums[i]
      array.push(nums[i]);
    }
  }
  return array.min();
};

// 返回当前数组的最小数
Array.prototype.min = function () {
  return Math.min(...this);
}
```
