# Leetcode 102. 二叉树的层序遍历

[102. 二叉树的层序遍历](https://leetcode-cn.com/problems/binary-tree-level-order-traversal/)


题解核心：广度优先搜索(Breadth First Search)

```
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
var levelOrder = function (root) {
  const ret = [];
  if (root == null) {
    return ret;
  }

  // 生成一个空的队列，用于存放节点
  const queue = [];
  queue.push(root); // 将跟节点推入队列中

  // 只要队列中有节点就开始处理
  while (queue.length > 0) {
    const currentLevelSize = queue.length; // 获取当前层级
    ret.push([]); // ret 数组推入一个新的数组，用于存储新的节点
    // 开始遍历层级，level 1 = 2个节点，level 2 = 4个节点，以此类推
    for (let i = 1; i <= currentLevelSize; i++) {
      const node = queue.shift(); // 取出队列中的第一个节点
      ret[ret.length - 1].push(node.val); // 将队列中的节点推入返回数组的最后一个子数组中

      if (node.left) {
        // 如果节点有左节点，推入队列中，等待下一次处理
        queue.push(node.left);
      }

      if (node.right) {
        // 如果节点有右节点，推入队列中，等待下一次处理
        queue.push(node.right);
      }
    }
  }
  return ret;
};
```
