# Leetcode 678 有效的括号字符串

```
/**
 * @param {string} s
 * @return {boolean}
 */
var checkValidString = function (s) {
  let min = 0;
  let max = 0;
  for (let i = 0; i < s.length; i++) {
    let current = s.charAt(i);
    if (current == '(') {
      min++;
      max++;
    } else if (current == ')') {
      if (min > 0) {
        min--;
      }
      max--;
    } else {
      if (min > 0) {
        min--;
      }
      max++;
    }
    if (max < 0) {
      return false;
    }
  }
  return min == 0;
};


console.log(checkValidString("(*))"));
```
